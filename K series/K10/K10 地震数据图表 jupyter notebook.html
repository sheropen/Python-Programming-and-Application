import csv
import matplotlib.pyplot as plt
import matplotlib
import numpy as np
with open("all_month.csv",encoding ="utf-8") as csvfile:
    reader = csv.DictReader(csvfile)
    depths,mags = [],[]
    for row in reader:
        try:
            depths.append(float(row['depth']))
        except:
            depths.append(0)
        try:
            mags.append(float(row['mag']))
        except:
            mags.append(0)
            
print(f"{len(mags)}")
print(f"{min(depths):.2f}",f"{max(depths):.2f}",
      f"{sum(depths)/len(depths):.2f}")
print(f"{min(mags):.2f}",f"{max(mags):.2f}",
      f"{sum(mags)/len(mags):.2f}")
